    <script>
        const correctAnswers = {
            1: 'B',
            2: 'D', 
            3: 'C',
            4: 'D'
        };
        
        let currentQuestion = 1;
        let score = 0;
        let answered = false;

        // Add click handlers to all option buttons
        document.querySelectorAll('.option-button').forEach(button => {
            button.addEventListener('click', function() {
                if (answered) return;
                
                const selectedOption = this.dataset.option;
                const questionNum = parseInt(this.closest('.question-card').dataset.question);
                const isCorrect = selectedOption === correctAnswers[questionNum];
                
                // Disable all buttons in this question
                const allButtons = this.closest('.question-card').querySelectorAll('.option-button');
                allButtons.forEach(btn => {
                    btn.style.pointerEvents = 'none';
                    if (btn.dataset.option === correctAnswers[questionNum]) {
                        btn.classList.add('correct');
                    } else if (btn === this && !isCorrect) {
                        btn.classList.add('incorrect');
                    }
                });
                
                // Show explanation
                this.closest('.question-card').querySelector('.explanation').classList.remove('hidden');
                
                // Update score
                if (isCorrect) {
                    score++;
                    // Add celebration effect
                    this.classList.add('celebration');
                }
                
                answered = true;
                
                // Show next button or results
                if (currentQuestion < 4) {
                    document.getElementById('next-btn').classList.remove('hidden');
                } else {
                    setTimeout(() => {
                        showResults();
                    }, 2000);
                }
                
                // Update progress
                updateProgress();
            });
        });

        function nextQuestion() {
            // Hide current question
            document.querySelector(`[data-question="${currentQuestion}"]`).classList.add('hidden');
            document.querySelector(`[data-question="${currentQuestion}"]`).classList.remove('active');
            
            // Show next question
            currentQuestion++;
            const nextCard = document.querySelector(`[data-question="${currentQuestion}"]`);
            nextCard.classList.remove('hidden');
            setTimeout(() => {
                nextCard.classList.add('active');
            }, 100);
            
            // Hide next button and reset answered state
            document.getElementById('next-btn').classList.add('hidden');
            answered = false;
        }

        function updateProgress() {
            const progress = (currentQuestion / 4) * 100;
            document.getElementById('progress-bar').style.width = progress + '%';
            document.getElementById('progress-text').textContent = `${currentQuestion}/4`;
        }

        function showResults() {
            document.getElementById('quiz-container').classList.add('hidden');
            document.getElementById('results').classList.remove('hidden');
            document.getElementById('final-score').textContent = `${score}/4`;
            
            // Add celebration animation
            document.getElementById('results').classList.add('celebration');
        }

        function restartQuiz() {
            // Reset variables
            currentQuestion = 1;
            score = 0;
            answered = false;
            
            // Reset UI
            document.getElementById('quiz-container').classList.remove('hidden');
            document.getElementById('results').classList.add('hidden');
            document.getElementById('next-btn').classList.add('hidden');
            
            // Reset all questions
            document.querySelectorAll('.question-card').forEach((card, index) => {
                if (index === 0) {
                    card.classList.remove('hidden');
                    card.classList.add('active');
                } else {
                    card.classList.add('hidden');
                    card.classList.remove('active');
                }
                
                // Reset buttons
                card.querySelectorAll('.option-button').forEach(btn => {
                    btn.style.pointerEvents = 'auto';
                    btn.classList.remove('correct', 'incorrect', 'celebration');
                });
                
                // Hide explanations
                card.querySelector('.explanation').classList.add('hidden');
            });
            
            // Reset progress
            updateProgress();
        }

        // Initialize progress
        updateProgress();
    </script>
